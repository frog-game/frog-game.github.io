		memcpy(pByteQueue->pBuffer, (const char*)pInBytes + nWriteBytes, nLength - nWriteBytes);//å‰©ä½™çš„åœ¨ç›´æ¥æ‹·è´

	pByteQueue->nWriteIndex = (pByteQueue->nWriteIndex + nLength) % pByteQueue->nCapacity;//é‡ç½®å†™ç´¢å¼•
	if (pByteQueue->nReadIndex == pByteQueue->nCapacity)

		pByteQueue->nReadIndex = 0;//é‡ç½®è¯»ç´¢å¼•
	}

	size_t nBytesToRead = nBytesWritten < nMaxLengthToRead ? nBytesWritten : nMaxLengthToRead;//å¾—åˆ°å¯è¯»å–çš„å¤§å°

	}

	{

		memcpy((char*)pOutBytes+nReadBytes,pByteQueue->pBuffer,nBytesToRead-nReadBytes);//è¯»å–å‰©ä½™éœ€è¦è¯»å–çš„å¤§å°

void byteQueue_reserve(byteQueue_tt* pByteQueue, size_t nCapacity)
